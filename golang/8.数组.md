package main

import "fmt"

func main() {
	// 定义数组
	var a = [5]int{1, 2, 3, 4, 5}
	var b = [3]string{"北京", "上海", "深圳"}
	fmt.Printf("%T:%d, %T:%s\n", a, a, b, b)

	// ... 用编译自动填写长度
	var c = [...]int{1, 2, 3, 3, 3, 3, 3, 3, 3}
	fmt.Printf("%T:%d\n", c, c)

	// 初始值
	var d = [20]int{19: 1}
	fmt.Println(d)

	// 数组的索引使用
	fmt.Println(d[19])

	// 遍历方式一
	for i := 0; i < len(a); i++ {
		fmt.Println(a[i])
	}
	// 遍历方式二
	for k, v := range b {
		fmt.Println(k, v)
	}

	// 练习：求数组的和
	list := [5]int{1, 3, 5, 7, 8}
	var sum int

	for i := 0; i < len(list); i++ {
		sum += list[i]
	}
	fmt.Println(sum)

}
