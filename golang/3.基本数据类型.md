

#### 1. 基本数据类型


```
package main

import (
	"fmt"
	"math"
)

func main() {
	var a int = 10
	var b int = 077
	var c int = 0xff

	fmt.Println(a, b)
	fmt.Printf("%b \n", a)
	fmt.Printf("%o \n", b)
	fmt.Printf("%x",c)

	// 求C的内存地址
	fmt.Printf("%p\n", &c)

	// 浮点数常量
	fmt.Println(math.MaxFloat64)
	fmt.Println(math.MaxFloat32)

	// 字符串转义
	fmt.Println("\"c:\\go\"")

	var s1 = "单行"
	var s2 = `
		这\
		是
		多
		行
	`
	fmt.Println(s1,s2)

	
}

```

#### 2. bytes and rune

-   uint8类型，或者叫 byte 型，代表了ASCII码的一个字符。
-   rune类型，代表一个 UTF-8字符。
```
	b1 := "big"
	b2 := []byte(b1)
	b2[0] = 'p'
	fmt.Println(b1,string(b2))

	r1 := "男人"
	r2 := []rune(r1)
	r2[0] = '女'
	fmt.Println(r1,string(r2))
```


```
package main

import "fmt"
import "math"
import "strings"


func main(){
	// 字符串方法
	s1 := "alexsb"
	fmt.Println(len(s1))

	// 拼接字符串
	s2 := "python"
	fmt.Println(s1 + s2)

	s3 := fmt.Sprintf("%s = %s", s1,s2)
	fmt.Println(s3)

	// 分割
	ret := strings.Split(s1, "x")
	fmt.Println(ret)

	// 判断是否包含
	s4 := strings.Contains(s1, "sb")
	fmt.Println(s4)

	// 判断前缀和后缀
	s5 := strings.HasPrefix(s1, "alex")
	s6 := strings.HasSuffix(s1, "sb")
	fmt.Println(s5,s6)

	// 求子串的位置
	s7 := "apple pen"
	fmt.Println(strings.Index(s7, "p"))
	fmt.Println(strings.LastIndex(s7, "p"))

	// join
	a1 := []string{"python", "php", "ruby"}
	fmt.Println(strings.Join(a1, "-"))
	
	// byte and rune
	s10 := "hello沙何"
	for i :=0; i < len(s10); i++ {
		fmt.Printf("%v (%c)\n", s10[i], s10[i])
	}
	fmt.Println()

	for k, v := range s10 {
		fmt.Printf("%v (%c)\n", k,v)
	}
	fmt.Println()

	// 修改字符串
	b1 := "big"
	b2 := []byte(b1)
	b2[0] = 'p'
	fmt.Println(string(b2))

	c1 := "您好"
	c2 := []rune(c1)
	c2[0] = '你'
	fmt.Println(string(c2))

	// 转换
	var a, b int = 3,4
	var c int
	c = int(math.Sqrt(float64(a*a + b*b)))
	fmt.Println(c)

	// 练习, 中英文遍历
	d1 := "hello沙河小王子"
	d2 := []rune(d1)

	fmt.Println(d2)
	for k,v := range d2 {
		fmt.Println(k, v)
	}

	// 练习,字符串反转
	q1 := "hello"
	list := []byte(q1)
	q2 := ""

	for i := len(list)-1; i>=0; i-- {
		q2 = q2 + string(list[i])
		fmt.Println(q2)

	}
}

```
